`; document.getElementById('wc-standings-body').innerHTML = `Unavailable`; } } function renderWcMatches(matches) { const container = document.getElementById('wc-matches'); const liveDot = document.getElementById('wc-live-dot'); // Sort: live first, then by date desc, show max 12 const sorted = [...matches].sort((a, b) => { const aLive = a.status === 'in_progress' || a.status === 'live'; const bLive = b.status === 'in_progress' || b.status === 'live'; if (aLive && !bLive) return -1; if (!aLive && bLive) return 1; return new Date(b.date || 0) - new Date(a.date || 0); }).slice(0, 12); const hasLive = sorted.some(m => m.status === 'in_progress' || m.status === 'live'); if (hasLive) liveDot.style.display = 'block'; if (!sorted.length) { container.innerHTML = '
No matches yet.

${home}
${scoreStr}
${metaStr}
${away}
`; container.appendChild(row); }); } function formatWcDate(dateStr) { if (!dateStr) return '–'; try { const d = new Date(dateStr); return d.toLocaleString(undefined, {month:'short', day:'numeric', hour:'2-digit', minute:'2-digit'}); } catch { return '–'; } } function renderWcGroupTabs(groups) { const tabsEl = document.getElementById('wc-group-tabs'); tabsEl.innerHTML = ''; // API may return array of groups or object — normalize const groupList = Array.isArray(groups) ? groups : Object.entries(groups).map(([name, data]) => ({name, ...data})); if (!groupList.length) { document.getElementById('wc-standings-body').innerHTML = `No group data yet.`; return; } // Store for tab switching groupList.forEach(g => { const name = g.name || g.group || g.letter || '?'; wcGroupsData[name] = g; }); const groupNames = Object.keys(wcGroupsData).sort(); wcActiveGroup = groupNames[0]; groupNames.forEach(name => { const btn = document.createElement('div'); btn.className = 'wc-group-tab' + (name === wcActiveGroup ?
